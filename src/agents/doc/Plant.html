def grow
        if @energy<@@maxSize
                stepGrowth=@@logisticRate*@energy*( (@@maxSize-@energy) / @@maxSize )
                if @energy+stepGrowth <= @@maxSize
                        @energy=@energy+stepGrowth
                else
                        @energy=@@maxSize
                end
        end
end
